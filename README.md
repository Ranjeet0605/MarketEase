Create a .env file in the root directory and add the required environment variables as shown in .env.example

# ===============================
# 🌐 SERVER CONFIGURATION
# ===============================
PORT=4000 || 3000

# ===============================
# 🗄️ DATABASE CONFIGURATION
# ===============================
DB_URI=mongodb://localhost:27017/ecommerce

# ===============================
# 🔐 JWT AUTHENTICATION
# ===============================
JWT_SECRET=your_jwt_secret_key_here
JWT_EXPIRE=5d
COOKIE_EXPIRE=5

# ===============================
# 💳 STRIPE PAYMENT CONFIGURATION
# ===============================
STRIPE_API_KEY=your_stripe_publishable_key
STRIPE_SECRET_KEY=your_stripe_secret_key

# ===============================
# 📧 EMAIL CONFIGURATION (SMTP)
# ===============================
SMTP_SERVICE=gmail
SMTP_MAIL=your_email@gmail.com
SMTP_PASSWORD=your_email_app_password
SMTP_HOST=smtp.gmail.com
SMTP_PORT=587

# ===============================
# ☁️ CLOUDINARY CONFIGURATION
# ===============================
CLOUDINARY_NAME=your_cloud_name
CLOUDINARY_API_KEY=your_cloudinary_api_key
CLOUDINARY_API_SECRET=your_cloudinary_api_secret

# ===============================
# 🔗 SOCIAL LINKS (FOR UI)
# ===============================
INSTAGRAM_URL=https://instagram.com/your_profile
LINKEDIN_URL=https://linkedin.com/in/your_profile
FACEBOOK_URL=https://facebook.com/your_profile
